class Solution {
public:
    stack<int> stack;

    bool isValid(string s) {
        for (int i = 0; i < s.length(); i++){//for each character in the string
            if     (s.at(i) == '[')stack.push('[');
            else if (s.at(i) ==']'){
                if(stack.top() != '[') return false;
                stack.pop();
            }
            else if (s.at(i) =='(')stack.push('(');
            else if (s.at(i) ==')'){
                if (stack.top() != '(') return false;
                stack.pop();
            }
            else if (s.at(i) =='{') stack.push('{');
            else if (s.at(i) =='}'){
                if (stack.top() != '{') return false;
                stack.pop();
            }
        }

        return true;
    }
};
